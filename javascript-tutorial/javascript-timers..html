navigator 객체

타이머

자바스크립트에서는 미래의 특정 시점에 코드를 실행할 수 있는데, window 객체에서는 다음과 같은 전역 함수를 제공한다.

  • setTimeout() 메서드는 특정 밀리초만큼의 시간이 경과한 후에 실행될 함수를 예약하며, 예약된 함수의 실행을 취소하기 위해 clearTimeout()에 전달할 수 있는 값을 반환한다.
  • clearTimeout() 메서드는 예약된 함수의 실행을 취소하는 데 사용된다.
  • setInterval() 메서드는 밀리초 단위의 특정 시간 간격에 걸쳐 반복적으로 실행될 함수를 지정한다. 이 함수는 예약된 함수가 나중에 호출되는 것을 취소하기 위해 clearInterval() 메서드에 전달할 수 있는 값을 반환한다.
  • clearInterval() 메서드는 예약된 함수의 호출을 취소하는 데 사용된다.

setTimeout()과 clearTimeout() 예제

setTimeout()과 clearTimeout()을 사용하는 예제는 다음과 같다.

var timeout;

function start(){
    timeout = setTimeout(function(){
        console.log("Hell world.");
    }, 3000);
}

function stop(){
    clearTimeout(timeout);
}

start();
stop();
setInterval() and clearInterval() example

var interval;
var count = 0;

function start(){
    interval = setInterval(function(){
        console.log("Hell world.");
        count++;
        if (count == 5) stop();
    }, 2000);
}

function stop(){
    clearInterval(interval);
}

start();
